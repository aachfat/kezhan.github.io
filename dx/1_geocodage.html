#### mcdo #####

url="https://www.restaurants.mcdonalds.fr/"
page = htmlTreeParse(getURL(url),useInternal = TRUE,encoding="utf8")
locs = unlist(xpathApply(page, '//ul[@class="large-services"]//li/a', xmlGetAttr,"href"))
locn=unique(locs)
mcdo=NULL
for (i in 1:94){
  url=paste0("https://www.restaurants.mcdonalds.fr/",locn[i])
  page = htmlTreeParse(getURL(url),useInternal = TRUE,encoding="utf8")
  noms= unlist(xpathApply(page, '//div[@class="desc adresse"]/a/h3', xmlValue ))
  adresse = unlist(xpathApply(page, '//div[@class="desc adresse"]/p', xmlValue ))
  mcdo=rbind(mcdo,cbind(noms,adresse))
}

save(dtm,file="data/mcdo.Rdata")
